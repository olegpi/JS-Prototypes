				String.prototype.getQueryString = function(key) {		
				   key = key.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
				   var regex = new RegExp("[\\?&]" + key + "=([^&#]*)");
				   var qs = regex.exec(this);
					
				   return qs[1];
				}
